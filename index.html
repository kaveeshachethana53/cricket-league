ක්‍රීඩකයින්

function addPlayer() { const name = document.getElementById("name").value; const age = document.getElementById("age").value; const role = document.getElementById("role").value; const details = document.getElementById("details").value; const photo = document.getElementById("photo").value; const msg = document.getElementById("msg"); msg.style.color = "red"; msg.innerHTML = ""; if (!name || !age || !role) { msg.innerHTML = "අත්‍යවශ්‍ය තොරතුරු පුරවන්න"; return; } fetch(WEB_APP_URL, { method: "POST", body: JSON.stringify({ name, age, role, details, photo }) }) .then(() => { msg.style.color = "green"; msg.innerHTML = "ක්‍රීඩකයා එකතු විය ✅"; loadPlayers(); }); document.getElementById("name").value = ""; document.getElementById("age").value = ""; document.getElementById("role").value = ""; document.getElementById("details").value = ""; document.getElementById("photo").value = ""; } function loadPlayers() { fetch(WEB_APP_URL + "?read=true") .then(res => res.json()) .then(data => { const playersDiv = document.getElementById("players"); playersDiv.innerHTML = ""; data.forEach(p => { playersDiv.innerHTML += `
නම: ${p.name}
වයස: ${p.age}
භූමිකාව: ${p.role}
විස්තර: ${p.details || "-"}
${p.photo ? `` : ""}
`; }); }); } window.onload = loadPlayers;
